# Потенциальные проблемы при 0.1 сек:
# 1. Для 50 страниц: 50 * 0.1 = 5 секунд (очень быстро)
# 2. Сервер может посчитать это атакой
# 3. Высокая вероятность получения 429 Too Many Requests
# 4. IP может быть временно заблокирован

# Пример кода с адаптивной задержкой
import time
import random

def adaptive_delay(base_delay=1):
    """
    Адаптивная задержка с случайной вариацией
    """
    # Добавляем случайность ±0.3 сек, чтобы не быть роботом
    jitter = random.uniform(-0.3, 0.3)
    delay = max(0.5, base_delay + jitter)  # Не меньше 0.5 сек
    time.sleep(delay)
