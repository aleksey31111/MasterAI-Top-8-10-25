def build_search_url(self, page: int = 1) -> str:
    # Способ 1: Базовая пагинация (для первой страницы)
    if page == 1:
        url = self.BASE_URL
    else:
        # Способ 2: Пагинация через /page/N/ (реальная структура сайта)
        url = f"{self.BASE_URL}page/{page}/"

    # Способ 3: GET-параметры для фильтрации (имитация поиска)
    params = {}
    if self.query:
        params['q'] = self.query
    if self.location:
        params['loc'] = self.location

    if params:
        # Способ 4: Использование urlencode для правильного кодирования
        url += "?" + urlencode(params)

    return url
