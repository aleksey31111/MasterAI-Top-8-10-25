import json
import os
from datetime import datetime


class ResumableBookScraper:
    def __init__(self, state_file='scraper_state.json', output_file='books.csv'):
        self.state_file = state_file
        self.output_file = output_file
        self.state = self.load_state()

    def load_state(self):
        """Загружает состояние парсера из файла"""
        if os.path.exists(self.state_file):
            try:
                with open(self.state_file, 'r') as f:
                    return json.load(f)
            except:
                return self.initial_state()
        return self.initial_state()

    def initial_state(self):
        """Начальное состояние парсера"""
        return {
            'last_page': 1,
            'processed_books': [],
            'total_books': 0,
            'last_update': None,
            'completed': False
        }

    def save_state(self):
        """Сохраняет текущее состояние"""
        self.state['last_update'] = datetime.now().isoformat()
        with open(self.state_file, 'w') as f:
            json.dump(self.state, f, indent=2)

    def is_book_processed(self, book_url):
        """Проверяет, была ли книга уже обработана"""
        return book_url in self.state['processed_books']

    def mark_book_processed(self, book_url):
        """Отмечает книгу как обработанную"""
        if book_url not in self.state['processed_books']:
            self.state['processed_books'].append(book_url)
            self.state['total_books'] += 1
